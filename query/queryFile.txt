SELECT COUNT(*)AS totalcard,playersId,playersName AS name,age,gender FROM Cards
JOIN Players ON Cards.playerId = Players.playersId
GROUP BY playersId
HAVING gender = 'Male'
ORDER BY totalcard DESC
LIMIT 3;
