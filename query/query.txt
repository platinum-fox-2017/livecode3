SELECT title,point,level,name,age
FROM Players INNER JOIN Status ON Players.id = Status.playerId
ORDER BY level DESC;

SELECT COUNT(*) as totalcard,Players.id as playerId, Players.name, Players.age, Players.gender
FROM Players JOIN Cards ON Players.id = Cards.playerId
WHERE Players.gender = 'Male'
GROUP BY Players.id
ORDER BY totalcard DESC LIMIT 3
